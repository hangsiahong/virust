```rust
use std::collections::HashMap;
use std::sync::Arc;
use virust_protocol::{HttpRequest, HttpResponse};
use serde_json::Value;

pub type HttpHandler = Arc<dyn Fn(HttpRequest) -> HttpResponse + Send + Sync>;
pub type WsHandler = Arc<dyn Fn(serde_json::Value) -> serde_json::Value + Send + Sync>;

#[derive(Clone)]
pub struct TypeDefinition {
    pub name: String,
    pub input_type: String,
    pub output_type: String,
}

pub struct RouteRegistry {
    http_routes: HashMap<String, HttpHandler>,
    ws_handlers: HashMap<String, WsHandler>,
    type_definitions: HashMap<String, TypeDefinition>,
}

impl RouteRegistry {
    pub fn new() -> Self {
        Self {
            http_routes: HashMap::new(),
            ws_handlers: HashMap::new(),
            type_definitions: HashMap::new(),
        }
    }

    pub fn register_http(&mut self, path: String, handler: HttpHandler) {
        self.http_routes.insert(path, handler);
    }

    pub fn register_ws(&mut self, path: String, handler: WsHandler) {
        self.ws_handlers.insert(path, handler);
    }

    pub fn register_type(&mut self, name: String, def: TypeDefinition) {
        self.type_definitions.insert(name, def);
    }

    pub fn get_http(&self, path: &str) -> Option<HttpHandler> {
        self.http_routes.get(path).cloned()
    }

    pub fn get_ws(&self, path: &str) -> Option<WsHandler> {
        self.ws_handlers.get(path).cloned()
    }

    pub fn generate_typescript(&self) -> String {
        let mut ts = String::from("// Auto-generated by Virust\n\n");

        for (_name, def) in &self.type_definitions {
            ts.push_str(&format!("export interface {} {{\n", def.input_type));
            ts.push_str(&format!("}};\n\n"));
            ts.push_str(&format!("export interface {} {{\n", def.output_type));
            ts.push_str(&format!("}};\n\n"));
        }

        ts
    }
}
```

```rust
use std::path::{Path, Buf};
use std::fs;
use walkdir::WalkDir;

pub struct RouteFile {
    pub path: String,
    pub file_path: Buf,
}

pub fn discover_routes(api_dir: &Path) -> Result<Vec<RouteFile>, std::io::Error> {
    let mut route_files = Vec::new();

    // Walk api/ directory for route.rs files
    for entry in WalkDir::new(api_dir)
        .follow_links(true)
        .into_iter()
        .filter_map(|e| e.ok())
    {
        let path = entry.path();

        // Only process route.rs files
        if path.file_name() == "route.rs" {
            // Extract URL path from file path
            let url_path = path.strip_prefix(api_dir)
                .unwrap()
                .to_str()
                .unwrap()
                .trim_start_matches('/');

            // Convert to URL path: api/chat/route.rs → /api/chat
            let normalized_path = url_path.replace("/route.rs", "");

            route_files.push(RouteFile {
                path: format!("/{}", normalized_path),
                file_path: path.to_path_buf(),
            });
        }
    }

    Ok(route_files)
}
```

```rust
use std::fs;
use std::process::Command;
use walkdir::WalkDir;

pub struct RouteFile {
    pub path: String,
    pub file_path: Buf,
}

pub fn discover_routes(api_dir: &Path) -> Result<Vec<RouteFile>, std::io::Error> {
    let mut route_files = Vec::new();

    // Walk api/ directory for route.rs files
    for entry in WalkDir::new(api_dir)
        .follow_links(true)
        .into_iter()
        .filter_map(|e| e.ok())
    {
        let path = entry.path();

        // Only process route.rs files
        if path.file_name() == "route.rs" {
            // Extract URL path from file path
            let url_path = path.strip_prefix(api_dir)
                .unwrap()
                .to_str()
                .unwrap()
                .trim_start_matches('/');

            // Convert to URL path: api/chat/route.rs → /api/chat
            let normalized_path = url_path.replace("/route.rs", "");

            route_files.push(RouteFile {
                path: format!("/{}", normalized_path),
                file_path: path.to_path_buf(),
            });
        }
    }

    Ok(route_files)
}

pub fn compile_routes(route_files: &[RouteFile]) -> Result<(), Box<dyn std::error::Error>> {
    // Compile as test harness to trigger macro registration
    for route_file in route_files {
        let crate_dir = route_file.file_path.parent()
            .ok_or(".".as_ref())?;

        // Run cargo check --test-harness on the file's directory
        let output = Command::new("cargo")
            .current_dir(crate_dir)
            .args(["check", "--test-harness", "-q"])
            .output()
            .await;

        if !output.status.success() {
            let stderr = String::from_utf8_lossy(&output.stderr);
            anyhow::anyhow!("Failed to compile {}: {}",
                route_file.path,
                stderr
            ))?;
        }
    }

    Ok(())
}
```

```rust
pub mod registry;
pub mod discovery;
pub mod http;
pub mod websocket;

use axum::{Router, routing::get};
use http::create_http_router;
use websocket::ws_upgrade;
use registry::RouteRegistry;
use discovery::{discover_routes, compile_routes};

pub struct VirustApp {
    registry: RouteRegistry,
    http_router: Router,
}

impl VirustApp {
    pub fn new() -> Self {
        let mut app = Self {
            registry: RouteRegistry::new(),
            http_router: Router::new(),
        };

        // Discover and register routes from api/ directory
        if let Ok(api_dir) = std::env::current_dir().map(|p| p.join("api")) {
            if api_dir.exists() {
                if let Ok(route_files) = discover_routes(&api_dir) {
                    let _ = compile_routes(&route_files);
                }
            }
        }

        // Setup HTTP routes
        app.setup_http_routes();

        app
    }

    fn setup_http_routes(&mut self) {
        self.http_router = create_http_router();
        self.http_router = self.http_router
            .route("/__types", get(types_handler))
            .route("/ws", get(ws_upgrade));
    }

    pub fn router(&self) -> Router {
        self.http_router.clone()
    }

    pub fn registry(&self) -> &RouteRegistry {
        &self.registry
    }
}

impl Default for VirustApp {
    fn default() -> Self {
        Self::new()
    }
}
```

```rust
    let input = parse_macro_input!(item as ItemFn);

    // Check if function name is an HTTP method
    let http_method = match fn_name.to_string().to_uppercase().as_str() {
        "GET" => Method::Get,
        "POST" => Method::Post,
        "PUT" => Method::Put,

            name: stringify!(#fn_name).to_string(),
            route_type: virust_macros::RouteType::Http(#http_method),
        });
    };

}
```

### Task 5.1: Define Persistence trait

**Files:**

- Modify: `crates/virust-protocol/src/lib.rs`

**Step 1: Write trait definition**

```rust
pub trait Persistence: Send + Sync {
    async fn get<T>(&self, id: &str) -> Result<Option<T>, PersistenceError>;
    async fn list<T>(&self) -> Result<Vec<T>, PersistenceError>;
    async fn create<T>(&self, item: T) -> Result<(), PersistenceError>;
    async fn update<T>(&self, id: &str, item: T) -> Result<(), PersistenceError>;
    async fn delete(&self, id: &str) -> Result<bool, PersistenceError>;
}

pub struct PersistenceError {
    pub kind: ErrorKind,
    pub message: String,
}

pub enum ErrorKind {
    NotFound,
    Serialization,
    Io,
}
```
